148      @Override 
149      public void loop() { 
150          DebrisMotor.setPower(.035); 
151          boolean ready = (RightHangServo.getPosition() == desiredPosition||LeftHangServo.getPosition()==desiredPosition); 
152          //drive code 
153          //may need change, this is from FTC 
154          if (gamepad1.left_stick_button||mountainMode) { 
155              float throttle = -gamepad1.left_stick_y; 
156              float direction = gamepad1.left_stick_x; 
157              float right = throttle - direction; 
158              float left = throttle + direction; 
159              right = Range.clip(right, -1, 1); 
160              left = Range.clip(left, -1, 1); 
161              right = smallPower * (float) scaleInput(right); 
162              left = smallPower * (float) scaleInput(left); 
163              motorRight.setPower(left); 
164              motorLeft.setPower(right); 
165   
166          } else { 
167              float throttle = -gamepad1.left_stick_y; 
168              float direction = gamepad1.left_stick_x; 
169              float right = throttle - direction; 
170              float left = throttle + direction; 
171              right = Range.clip(right, -1, 1); 
172              left = Range.clip(left, -1, 1); 
173              right = (float) scaleInput(right); 
174              left = (float) scaleInput(left); 
175              motorRight.setPower(left); 
176              motorLeft.setPower(right); 
177          } 
178   
179          //lift servo code 
180          //this is a continuous rotation servo 
181          //for a continuous rotation servo, 0 is spin one direction full power, 1 is the other direction full power 
182          //and 0.5 is stopped. 
183          /*if (gamepad2.dpad_up) { 
184              LiftServo.setPosition(0); 
185          } else if (gamepad2.dpad_down) { 
186              LiftServo.setPosition(1); 
187          } else { 
188              LiftServo.setPosition(0.5); 
189          }*/ 
190   
191              if (gamepad2.dpad_up) { 
192                      SearchingUp = true; 
193              } 
194              else if (gamepad2.dpad_down) { 
195                      SearchingDown = true; 
196              } 
197   
198   
199   
200   
201          //Bucket Tilting code 
202          //joystick reads from -1 to 1 on each axis 
203          float position = gamepad2.right_stick_x; 
204          //changing max angle allowed from degrees to encoder units 
205          float EncoderMax = (maxAngle / 360) * EncoderPerRotation60; 
206          //scaling position to have the max value as the max angle 
207          int TargetEncoderValue = (int) (position * EncoderMax); 
208          //DebrisMotor.setMode(DcMotorController.RunMode.RUN_TO_POSITION); 
209          if (gamepad2.right_stick_x > .1 || gamepad2.right_stick_x < -.1) { 
210              DebrisMotor.setTargetPosition(TargetEncoderValue); 
211          } 
212          if (gamepad2.right_stick_x <= .1 && gamepad2.right_stick_x >= -.1) { 
213              DebrisMotor.setTargetPosition(0); 
214          } 
215   
216   
217          //Roller Code 
218          //when trigger is pressed, power is set 
219          if (gamepad2.right_trigger > triggerCutoff) { 
220              RollerMotor.setPower(power); 
221          } else if (gamepad2.left_trigger > triggerCutoff) { 
222              RollerMotor.setPower(-power); 
223          } else { 
224              RollerStop(); 
225          } 
226   
227   
228          //Climber code 
229          if (gamepad1.right_bumper) { 
230              flipClimbers(); 
231              climbersFlipped = true; 
232          } else if(climbersFlipped) { 
233              ClimberServo.setPosition(0.3); 
234          }else { 
235              ClimberServo.setPosition(0); 
236          } 
237  if(gamepad2.left_bumper){ 
238      RightHangServo.close(); 
239      LeftHangServo.close(); 
240  } 
241  //Hopper Door code 
242          if ((Math.abs(DebrisMotor.getCurrentPosition()) * 360 / EncoderPerRotation60) >= minAngle) { 
243              OpenDoors(); 
244          } else { 
245              HopperDoor.setPosition(.5); 
246          } 
247   
248          //Braking code 
249          if (gamepad1.y) { 
250              Brake(); 
251   
252          } else if(gamepad2.left_stick_x>0){ 
253              RightZipline.setPosition((gamepad2.left_stick_x/2)+.5); 
254          }else if(gamepad2.left_stick_x<0){ 
255              LeftZipline.setPosition((-gamepad2.left_stick_x/2)+.5); 
256          }else if (gamepad2.left_stick_x==0){ 
257              LeftZipline.setPosition(.5); 
258              RightZipline.setPosition(.5); 
259          }else { 
260              UnBrake(); 
261      } 
262   
263   
264  if(SearchingUp&&!OpenFound){ 
265       LiftServo.setPosition(0); 
266      FindOpen(); 
267   
268  } 
269          if(SearchingUp&&OpenFound) { 
270              LiftServo.setPosition(0); 
271              LookForClip(); 
272          } 
273   
274          if(SearchingDown&&!OpenFound){ 
275              LiftServo.setPosition(1); 
276              FindOpen(); 
277          } 
278          if(SearchingDown&&OpenFound){ 
279              LiftServo.setPosition(1); 
280              LookForClip(); 
281          } 
282  if(SearchingDown&&SearchingUp){ 
283      SearchingDown = false; 
284  } 
285   
286   
287          //tails 
288          if(gamepad1.a){ 
289              mountainMode = true; 
290          } 
291          if(gamepad1.x){ 
292              mountainMode = false; 
293          } 
294  if(mountainMode) { 
295      RightTail.setPosition(1); 
296      LeftTail.setPosition(1); 
297  } else{ 
298      RightTail.setPosition(.15); 
299      LeftTail.setPosition(.15); 
300  } 
301          //hangers 
302          if(gamepad1.dpad_up&&hangerPosition !=2){ 
303              hangerPosition++; 
304          } 
305          if(gamepad1.dpad_down&&hangerPosition !=0){ 
306              hangerPosition--; 
307          } 
308          if(hangerPosition == 0){ 
309              desiredPosition = zero; 
310          } 
311          if(hangerPosition == 1){ 
312              desiredPosition = mid; 
313          } 
314          if(hangerPosition == 2){ 
315              desiredPosition = high; 
316          } 
317          RightHangServo.setPosition(desiredPosition+.01); 
318          LeftHangServo.setPosition(desiredPosition+offset); 
319   
320          if(gamepad1.right_trigger>triggerCutoff){ 
321              LeftHangMotor.setPower(.8); 
322              RightHangMotor.setPower(.8); 
323   
324          }else if(gamepad1.left_trigger>triggerCutoff) { 
325              LeftHangMotor.setPower(-.8); 
326              RightHangMotor.setPower(-.8); 
327          }else{ 
328              LeftHangMotor.setPower(0); 
329              RightHangMotor.setPower(0); 
330          } 
331          if(gamepad1.back){ 
332              hanging=true; 
333          } 
334   
335   
336          if(hanging){ 
337              LeftHangServo.setPosition(mid+offset); 
338              RightHangServo.setPosition(mid+0.01); 
339          } 
340   
341          //telemetry section 
342          telemetry.addData("elevatorGate", elevatorPhotogate.getValue()); 
343          telemetry.addData("elevatorEncoder", DebrisMotor.getCurrentPosition()); 
344   
345      }

352      /* 
353       * This method scales the joystick input so for low joystick values, the 
354       * scaled value is less than linear.  This is to make it easier to drive 
355       * the robot more precisely at slower speeds. 
356       */ 
357      double scaleInput(double dVal)  { 
358          double[] scaleArray = { 0.0, 0.05, 0.09, 0.10, 0.12, 0.15, 0.18, 0.24, 
359                  0.30, 0.36, 0.43, 0.50, 0.60, 0.72, 0.85, 1.00, 1.00 }; 
360   
361          // get the corresponding index for the scaleInput array. 
362          int index = (int) (dVal * 16.0); 
363   
364          // index should be positive. 
365          if (index < 0) { 
366              index = -index; 
367          } 
368   
369          // index cannot exceed size of array minus 1. 
370          if (index > 16) { 
371              index = 16; 
372          } 
373   
374          // get value from the array. 
375          double dScale = 0.0; 
376          if (dVal < 0) { 
377              dScale = -scaleArray[index]; 
378          } else { 
379              dScale = scaleArray[index]; 
380          } 
381   
382          // return scaled value. 
383          return dScale; 
384      }
